# list
$ ls

# long listing format
$ ls -l

# all files and directories including hidden
$ ls -a

$ ls -l -a

$ ls -la

$ ls -l /var/log

# human readable format: K[byte]

$ ls -lh

-----------------------------------------------------------------------------------------------------

# print working directory
$ pwd

# change directory
$ cd /var/log

# go to parent directory
$ cd ..

# go to root directory
$ cd /

# go to previous directory
$ cd -
